#include <iostream>
using namespace std;

int vremraz(int h1, int m1, int s1, int h2, int m2, int s2)
{
	int hus;
	int mus;
	int sus;
	if (h1 <= h2)
	{
		if (m1 <= m2)
		{
			if (s1 <= s2)
			{
			}
			else
			{
				hus = h1 * 3600 - h2 * 3600 + 3600 * 24;
				mus = m1 * 60 - m2 * 60 + 60;
				sus = s1 - s2;
				return hus + mus + sus;
			}
		}
		else
		{
			hus = h1 * 3600 - h2 * 3600 + 3600*24;
			mus = m1 * 60 - m2 * 60;
			sus = s2 - s1;
			return hus + mus + sus;
		}
	}
	else 
	{
		hus = h2 * 3600 - h1 * 3600;
		mus = m2 * 60 - m1 * 60;
		sus = s2 - s1;
		return hus + mus + sus;
	}
}


int main()
{
	int h1, m1, s1, h2, m2, s2;

}
