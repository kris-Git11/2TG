#include <iostream>
using namespace std;


int main()
{
	int a, b;
	int brojac;
	char c;
	cout << "upisi prvi broj";
	cin >> a;
	cout << "upisi drugi broj";
	cin >> b;
	cout << "\n";
	cout << "Upisi znak za zeljenu operaciju\n";
	cout << "(a - zbroj)\n)";
	cout << "(b - broj veci od ta dva broja)\n";
	cout << "(c - aritmeticka i geometrijska sredina)\n";
	cout << "(d - svi neparni brojevi izmedu ta dva broja)\n";
	cout << "(e - najmanji zajednicki visekratnik)\n";
	cin >> c;
	switch (c)
	{
	case ('a'):
	{
		int zbroj;
	zbroj = a + b;
	cout << "Zbroj je " << zbroj << endl;
	}

	case('b'):
	{
		int broj;
			if (a >= b)
			{
				broj = a + 1;
			}
			else
			{
				broj = b + 1;
			}
			cout << "broj koji je veci od oba broje je " << broj << endl;
	}
	case('c'):
	{
		double sredina;
		sredina = (a + b) / 2;
		cout << "Aritmeticka i geometrijska sredina je " << sredina << endl;
	}
	case('d'):
	{
		if (a < b)
		{
			for (int i = 1; i <= (b-a)/2; i++)
			{
				int j = 1;
				if (a % 2 == 0)
				{
					brojac = a + 2 * i - 1;
					cout << brojac << " ";
				}
				else
				{
					brojac = a + 2 * i;
					cout << brojac << " ";
				}
				if (j % 5 == 0)
				{
					cout << "\n";
				}
				j = j + 1;
			}
		}
		else
			for (int i = 1; i <= (a-b)/2; i++)
			{
				int j = 1;
				if (a % 2 == 0)
				{
					brojac = b + 2 * i - 1;
					cout << brojac << " ";
				}
				else
				{
					brojac = b + 2 * i;
					cout << brojac << " ";
				}
				if (j % 5 == 0)
				{
					cout << "\n";
				}
				j = j + 1;
			}
			
	}
	case ('e'):
	{
		int visekratnik;
		for (int i = 1; i <= a; i++)
		{
			if (b * i % a == 0)
			{
			visekratnik = i * b;
				break;
			}

		}
		cout << visekratnik;
	}
	}
}
